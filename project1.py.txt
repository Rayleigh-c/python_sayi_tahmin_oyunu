from random import randint

print("Merhaba sayi tahmin oyunumuza hos geldiniz 😍")
print("Bilgisayar 1 ile 100 arasinda bir sayi tuttu! 🤖")
print("Bakalim kac denemede dogru sayiyi tahmin edebileceksiniz 🤭")


x = randint(1, 100)



deneme = 0

while True:
    try:
        a = int(input("Lutfen tahmin ettiginiz sayiyi giriniz 🙌 : "))
    except ValueError:
        print("Lutfen sadece sayi giriniz 😓")

    deneme += 1

    if a < x:
        print("Daha buyuk bir sayi giriniz 😒")
    elif a > x:
        print("Daha kucuk bir sayi giriniz 👎")
    else:
        print(f"Tebrikler {deneme}. denemede dogru sayiyi buldunuz! 🎉")
        break
