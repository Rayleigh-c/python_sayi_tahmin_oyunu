#BANKA SIMULATION

def bakiyeGoruntule(bakiye):
    print(f"Hesabinizin guncel bakiyesi : {bakiye:.2f} TL\n")
    return bakiye                                #Bakiye parametresi fonkisyon disinda da deger dondurebilmesi icin bakiye degiskenini return ile disari dondurdum, aynisi asagidaki diger fonksiyonlar icin de gecerli

def paraYatir(bakiye):
    try:
        yatir = float(input("Lutfen yatirmak istediginiz tutari giriniz : "))
        if yatir <= 0:                            #Kullanici eğer negatif bir tutar yatirirsa mantık hatasi olur ve kod coker bunun olmamasi adina try-except koydum
            print("Lutfen gecerli tutar giriniz!")
        else:
            bakiye += yatir
            print("Hesabinizin guncel bakiyesi : ", bakiye)
    except ValueError:
        print("Yanlis tutar girdiniz !")
    return bakiye

def paraCekme(bakiye):
    print("Hesabinizdaki bakiye : ", bakiye)
    try:
        cekme = float(input("Lutfen cekmek istediginiz tutari giriniz : "))
        if cekme > bakiye:                         #Mantik hatasi olmamasi adina eğer kullanici bakiyesinden fazla tutar cekmek isterse hata mesaji isle karsilasiyor
            print("Yetersiz bakiye!")
        elif cekme <= 0:                           #Kullanici eğer negatif bir tutar cekmek isterse hata mesaji ile karsilasiyor
            print("Lutfen gecerli tutar giriniz!")
        else:
            bakiye -= cekme
        print("Hesabinizin guncel bakiyesi : ", bakiye)
    except ValueError:
        print("Yanlis tutar girdiniz !")
    return bakiye
